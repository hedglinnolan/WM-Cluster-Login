# Save as C:\Users\<windows_username>\.ssh\config   (create the folder if needed)
# Replace <YOUR_NETID> with your W&M username.
Host wm-bastion
  HostName bastion.wm.edu
  User <YOUR_NETID>

Host wm-astral
  HostName astral.sciclone.wm.edu
  User <YOUR_NETID>
  ProxyJump wm-bastion
